\section{From eq.~(33) to the quartic, and the $\kz=0$ anchor}

\begin{frame}{Clearing eq.~(33) to a quartic}
Substituting the harmonic-core definitions into eq.~(33) and writing
$C\equiv(2n+1)\sqrt{\alpha^3/2}\,\Vo$:
\[
\omega^2-\kz^2-\frac{\alpha^2\Vo\kz}{\omega^2}=\frac{C}{\omega}
\]
Multiply through by $\omega^2$ ($\omega=0$ is not a root) and collect:
\[
\boxed{\;\omega^4-\kz^2\,\omega^2-C\,\omega-\alpha^2\Vo\kz=0\;}
\]
the \textbf{WKB quartic}. $\gamma=\max\Imy\,\omega$ over its four roots is the
``$\gamma_{\rm WKB}$'' quoted in every early program table.
\begin{caveat}
An early implementation assumed $\Rey\,\omega=0$ (purely growing) and bisected on
real $\gamma$ — wrong away from $\kz=0$ by up to 50\% at $\kz=0.25$. The full complex
quartic must be solved.
\end{caveat}
\end{frame}

\begin{frame}{The $\kz=0$ limit: chromo-Weibel growth rate}
\footnotesize
Set $\kz=0$: $\omega^4-C\omega=0\Rightarrow\omega(\omega^3-C)=0$. The growing root
is $\omega=C^{1/3}e^{2\pi\ii/3}$, giving
\[
\boxed{\;\gamma_{\rm Weibel}(\kz{=}0)=\Bigl(\sqrt{\tfrac{\alpha^3}{2}}\,\Vo\Bigr)^{1/3}\sin 60^\circ\;}
\]
\begin{physpicture}
\textbf{Not} a bunching instability — nothing varies along $z$. It's
\emph{filamentation}: $\Az$ rotates color-2/3 fields into each other while
counter-streaming currents feed them back. Closing the loop $E\to A\to Q\to
J\to E$ gives gain $\propto C/\gamma^3$; marginal self-amplification fixes
$\gamma\sim C^{1/3}$, root geometry supplies $\sin60^\circ$.
\end{physpicture}
\textbf{Validated to 0.5\%}: $\alpha=2,\Vo=0.1$: $\gamma_{\rm sim}=0.5039$ vs.\
formula $0.5065$, over nine decades of growth — the strongest single anchor of
the quartic, and the first number quoted to a referee (parameter-free).
\end{frame}

\begin{frame}{Quartic vs.\ reality: where it's valid, where it fails}
\includegraphics[width=0.4\textwidth]{fig_quartic.pdf}
\begin{itemize}
\item \textbf{$\kz=0$}: exact, validated to 0.5\%.
\item \textbf{Low $\kz$}: the quartic's drive $\alpha^2\Vo\kz/\gamma^2$ has no

\item \textbf{High $\kz$}: quartic balance gives $\gamma^2\approx\alpha^2\Vo/\kz$,

\end{itemize}
The diagnosis (why, precisely) has to wait for the exact reduction — coming up.
\end{frame}

\section{The time-domain model}

\begin{frame}{The simulated system}
For each color $a\in\{1,2,3\}$: fields $E_x^a,E_z^a,B_y^a,A_z^a$. For each beam
$s\in\{A,B\}$: fluid $n_s,p_{xs},p_{zs}$, color charge $Q_s^a$.
\vspace{0.05em}
\begin{align*}
&\partial_t n+\nabla\!\cdot(n\vec v)=0, \qquad
\partial_t p_x+\nabla\!\cdot(\vec v p_x)=-n\textstyle\sum_a Q^a(E_x^a+v_zB_y^a)\\
&\partial_t p_z+\nabla\!\cdot(\vec v p_z)=-n\textstyle\sum_a Q^a(E_z^a-v_xB_y^a),\qquad
\partial_t Q^a=\alpha\,v_z(\vec Q\times\vec A_z)^a\\[0.3em]
&\partial_t E_x^a=-\partial_zB_y^a-\alpha(\vec A_z\times\vec B_y)^a-J_x^a,\qquad
\partial_t E_z^a=\partial_xB_y^a-J_z^a\\
&\partial_t B_y^a=\partial_xE_z^a-\partial_zE_x^a+\alpha(\vec A_z\times\vec E_x)^a,
\qquad \partial_t A_z^a=-E_z^a
\end{align*}
Beams: $v_{zA}=+\Vo\tanh\xi$, $Q^1_A=+1$; $v_{zB}=-\Vo\tanh\xi$, $Q^1_B=-1$; frozen
background $\Az(\xi)=-\Vo\ln\cosh\xi$, $u(\xi)\equiv-\alpha\Az(\xi)\ge0$.
\end{frame}

\begin{frame}{A gap worth closing: gauge counting}
\footnotesize
SU(2) gauge freedom eliminates exactly \emph{one} full spacetime component of
$A_\mu^a$ — not two. Dropping both $A_x^a,A_y^a$ while also fixing $A_0^a$ needs
justification, not assumption.
\[
\boxed{\text{one gauge choice}\Rightarrow\text{one component eliminated --- not two.}}
\]
\begin{itemize}
\item \texttt{khaxn.pdf} spends its choice on $a_x^a=0$; $\phi$ survives, killed
only by the Gauss \emph{constraint}; $a_y$ survives, decoupled from $a_z$ by
construction.
\item This model spends a \emph{different} choice, $A_0^a\equiv0$ (temporal
gauge — exactly what $\partial_tA_z=-E_z$ states).
\item $A_y$ is then a \emph{sector} choice (never populated); $A_x$ is wired to
\emph{nothing} in any equation of motion — a complete spectator either way.
\end{itemize}
\begin{physpicture}
One knob, one turn: two different single choices, not one spent twice. $E_x$
carries exactly $\phi$'s slaved role ($\gamma e_x=-\ii\OmA b$, purely algebraic).
\end{physpicture}
\end{frame}

\begin{frame}{Precession source from Wong's equation}
\footnotesize
Wong's equation, non-relativistic limit, temporal gauge, only $A_z^b\neq0$:
\[
\frac{\dd Q^a}{\dd t}=-g\,\epsilon^{abc}v_zA_z^b Q^c
=+\alpha\,v_z\,(\vec Q\times\vec A_z)^a
\]
exactly the model's source; $|\vec Q|$ conserved $\Rightarrow$ $Q^1_{A,B}=\pm1$
stays a unit adjoint vector.
\vspace{0.1em}
\textbf{Energy budget} (derivation skipped): non-Abelian cross terms do \emph{not}
cancel —
\[
\frac{\dd E}{\dd t}\Big|_{\rm NA}=-2\alpha\int\vec A_z\cdot(\vec B_y\times\vec E_x)
\]
\begin{physpicture}
A frozen field is an infinite battery. In the tachyonic region this term
\emph{feeds} the wave without limit (Nielsen--Olesen-type decay). In the shear
well it's a drain; the shear branch draws on the beams' kinetic energy instead —
a finite, self-consistent reservoir.
\end{physpicture}
\end{frame}

\section{Six-field linearization}

\begin{frame}{Circular pairs: color rotation is a phase}
Combine color-2/3 perturbations: $b=B_y^2+\ii B_y^3$, $e_x=E_x^2+\ii E_x^3$,
$e_z=E_z^2+\ii E_z^3$, $a=A_z^2+\ii A_z^3$, $q_{A,B}=Q^2_{A,B}+\ii Q^3_{A,B}$, ansatz
$\propto e^{\ii\kz z+\gamma t}$. For background $\vec A_z=(\Az,0,0)$, any
commutator $(\vec A_z\times\vec X)^2+\ii(\vec A_z\times\vec X)^3=\ii\Az X^c$: a
cross product with the background acts on a circular pair as multiplication by
$\ii\Az$.
\[
\boxed{\;\OmA(\xi)=\kz+\alpha\Az=\kz-u(\xi),\qquad \OmF(\xi)=\kz-\alpha\Az=\kz+u(\xi)\;}
\]
\begin{physpicture}
The color-2/3 waves are \emph{charged} (charge $\pm1$) under the residual $U(1)$
about the color-1 axis; $\alpha\Az$ shifts their momentum like an ordinary gauge
potential, $\kz\to\kz\pm\alpha\Az$. Amp\`ere sees $\OmA$, Faraday sees $\OmF$ — the
split is the fingerprint of the fixed $A_x=0$ constraint.
\end{physpicture}
\end{frame}

\begin{frame}{The six-field eigenvalue system}
\footnotesize
\[
\boxed{
\begin{aligned}
\gamma\,b &= -\ii\,\OmF\,e_x+e_z' & \gamma\,a &= -e_z\\
\gamma\,e_x &= -\ii\,\OmA\,b & \gamma\,q_A &= \ii\alpha v\,a-\ii v\,\OmA\,q_A\\
\gamma\,e_z &= b'+v\,(q_A-q_B) & \gamma\,q_B &= \ii\alpha v\,a+\ii v\,\OmA\,q_B
\end{aligned}}
\]
\normalsize
\vspace{0.1em}
This is exactly what \texttt{ym\_eigenmode.py} solves. Validated against the 2D
simulation at 1--4\% on the $\Vo=0.05$ validation stack ($\kz\le5$); 4--9\%
median across the full integer-$\kz$ map; 6--10\% archive-wide clean core.
\vspace{0.1em}
\textbf{Correspondence check}: eliminating $e_x,e_z,q_A,q_B,b$ reproduces exactly
the beam-response denominators $\Omega\pm\alpha\Phi_0^1$ of the boxed equation on
slide 2, under $\omega=\ii\gamma$ — the time-domain model linearizes into
\emph{precisely} the given frequency-domain theory, modulo two stated scope
truncations (no background $\phi_0^1$; no $a_y^\pm$ sector).
\vspace{0.1em}
\textbf{Symmetry}: helicity conjugation ($\Az\to-\Az$) combined with $\xi\to-\xi$
maps the system onto itself — every eigenmode on one side of the layer has a
mirror partner of the opposite polarization on the other side, same $\gamma$.
\end{frame}

\section{Local dispersion: the tachyonic branch and the drive branch}

\begin{frame}{Block factorization at $k_x=0$}
Freezing the background locally, the system splits into two independent blocks:
\[
\begin{gathered}
\underbrace{\gamma b=-\ii\OmF e_x,\;\; \gamma e_x=-\ii\OmA b}_{\text{EM block}}\\[5pt]
\underbrace{\gamma e_z=v(q_A-q_B),\;\gamma a=-e_z,\;\gamma q_{A,B}=\ii\alpha va\mp\ii v\OmA q_{A,B}}_{\text{charge block}}
\end{gathered}
\]
\end{frame}

\begin{frame}{EM block: the tachyonic (Nielsen--Olesen) branch}
\footnotesize
Multiplying the EM block:
\[
\boxed{\;\gamma^2=-\OmA\OmF=\alpha^2A^2-\kz^2\;}
\]
Negative wherever the background shift exceeds the wavenumber $\Rightarrow$
$\gamma^2>0$: the charged-vector-on-strong-background instability of the
Nielsen--Olesen class.
\[
\gamma_{\rm loc}(\xi)\simeq\sqrt{(\alpha\Vo)^2(|\xi|-\ln2)^2-\kz^2}
\quad\text{real for }
\boxed{\;|\xi|>\xi_{\rm crit}\simeq\frac{\kz}{\alpha\Vo}+\ln2\;}
\]
\centering\includegraphics[width=0.36\textwidth]{fig_tachyon.pdf}
\end{frame}

\begin{frame}{Consequence: high-$\alpha$ window collapse}
A window of radius $\xi_w$ excludes the tachyon only if $\xi_w<\xi_{\rm crit}$. As
$\alpha$ grows at fixed $\kz,\Vo$: $\xi_{\rm crit}\to\ln2$ — eventually \emph{no}
window both contains the shear well and excludes the tachyon.
\begin{caveat}
This is the mechanism of the early tachyonic blow-ups at $\alpha\ge6$: at
$\alpha=6,\Vo=0.05,\kz=2$: $\xi_{\rm crit}=7.4$, already inside the vetted
windows. The failure maps to $\alpha$, exactly as predicted — this is now known
to be only \emph{part} of the story (resolution addendum, later).
\end{caveat}
Same mechanism, evaluated at layer scale ($|\Az|\sim\Vo$) rather than the outer
plateau, gives the ``precession cascade'' $\gamma_{\rm casc}\approx\alpha\Vo$ that
motivated full eigenmode seeding (later in this talk).
\end{frame}

\begin{frame}{Charge block: the local drive branch}
Solving the two precession equations and closing through $e_z,a$:
\[
q_A-q_B=\frac{2\alpha v^2\OmA}{\gamma^2+v^2\OmA^2}\,a
\qquad\Longrightarrow\qquad
\boxed{\;\gamma^2=-g,\qquad g(\xi;\gamma)=\frac{2\alpha v^3\OmA}{\gamma^2+v^2\OmA^2}\;}
\]
\begin{physpicture}
Two driven pendulums. The potential $a$ shakes each beam's color charge; each
precesses at Doppler-shifted $\pm v\OmA$. Only their \emph{difference} radiates
back into $a$ through the current — growth is a phase-locking resonance between
the wave and the beams' color precession. This is the local ($k_x\to0$) limit of
the drive that reappears, unmodified, in the exact reduced ODE next.
\end{physpicture}
\end{frame}

\section{Exact scalar reduction: one ODE for $a(x)$}

\begin{frame}{The reduction — algebra all the way down}
\footnotesize
The six-field system has only \emph{one} $x$-derivative chain
($e_z'\to b\to b'$); everything else is algebraic. Eliminating $e_x,e_z,q_A,q_B,b$:
\[
b=-\frac{\gamma^2a'}{\D},\qquad \D(x)\equiv\gamma^2+\OmA\OmF=\gamma^2+\kz^2-u(x)^2
\]
Assembling Amp\`ere-$z$ gives \emph{exactly one} second-order ODE — the new
theory's counterpart of the boxed equation on slide 2, now derived independently
and exactly from the six-field system:
\[
\boxed{\;\Bigl(\frac{\gamma^2 a'}{\D}\Bigr)'=(\gamma^2+g)\,a,\qquad
g(x)=\frac{2\alpha v^3\OmA}{\gamma^2+v^2\OmA^2}\;}
\]
\begin{physpicture}
Three powers of $v$ in $g$: the wave must (i) shake the charges, (ii) make the
two beams respond \emph{differently} (opposite Doppler rotation), (iii) collect
the response as a current. The drive dies as $v^3$ toward the layer center — the
shear branch is anchored to the flow, not the field maximum.
\end{physpicture}
\end{frame}

\section{The drive, the resonance, and the growth ceiling}

\begin{frame}{The saturated drive $G(w)$ and its resonance}
In the saturated region ($v=\pm\Vo$, $u=\alpha\Vo(|\xi|-\ln2)$), on the driven
side:
\[
-g=G(w)=\frac{2\alpha\Vo^3w}{\gamma^2+\Vo^2w^2},\qquad w\equiv\OmA=\kz-u(\xi)
\]
Maximizing over $w$: $\dfrac{dG}{dw}=0\Rightarrow w^*=\gamma/\Vo$ — the mode grows
fastest where the \emph{beam-frame precession frequency} matches the growth rate
itself. Peak value:
\[
G_{\max}=G(w^*)=\frac{\alpha\Vo^2}{\gamma}
\]
\end{frame}

\begin{frame}{The growth ceiling: $\gamma^3\le\alpha\Vo^2$}
A confining well needs $G>\gamma^2$ somewhere; since $G\le G_{\max}=\alpha\Vo^2/\gamma$:
\[
\boxed{\;\gamma^3\le\alpha\Vo^2\;}
\]
independent of $\kz$, $\eps$, and window geometry. \textbf{Two independent
derivations agree} (well-existence, and the local-drive branch
$\gamma^2=G(w^*)$ solved directly).
\begin{physpicture}
A self-limiting resonance. The beam response is a Lorentzian in local precession
frequency, width set by $\gamma$ itself: the faster the mode grows, the broader
the band it averages over, the weaker the peak drive it can extract. The ceiling
is the fixed point — the mode grows exactly as fast as the drive it leaves
itself.
\end{physpicture}
\textbf{Measured}: dominant-branch peaks reach 95--99\% of $(\alpha\Vo^2)^{1/3}$ in
\emph{every} validated series (empirical confirmation later, T2.5).
\end{frame}

\begin{frame}{Natural units: a universal well}
\footnotesize
$\Ceil\equiv(\alpha\Vo^2)^{1/3}$ (rate scale), $\Kwn\equiv(\alpha/\Vo)^{1/3}$
(wavenumber scale), $\gamma=\Ceil\hat\gamma$. The drive becomes \emph{universal}
— no $\alpha,\Vo,\eps$ remain:
\[
G(w)=\Ceil^2\,\hat G(\hat w),\qquad \hat G(\hat w)=\frac{2\hat w}{\hat\gamma^2+\hat w^2}
\]
\centering\includegraphics[width=0.5\textwidth]{fig_drive.pdf}
\raggedright
Well edges $\hat w_\pm=(1\pm\sqrt{1-\hat\gamma^6})/\hat\gamma^2$, real only for
$\hat\gamma\le1$ — a \emph{third} derivation of the ceiling. \textbf{Rates scale
as $(\alpha\Vo^2)^{1/3}$, wavenumbers as $(\alpha/\Vo)^{1/3}$ — no function of
$\alpha\Vo$ alone works.}
\end{frame}

\section{The dispersion curve: confinement-set wavelength selection}

\begin{frame}{No intrinsic $\kz$ maximum}
In an unbounded domain, nothing stops the mode following the resonance surface
outward as $\kz$ grows: $G_{\max}=\alpha\Vo^2/\gamma$ is independent of $\kz$, so
$\gamma(\kz)$ rises monotonically to the ceiling while the mode migrates outward.
\begin{center}
\textbf{Directly demonstrated}: $\gamma$ climbs to $0.9955\,\Ceil$ by $\kz=10$
with \emph{no turnover}, mode peak at $\xi=-36$ (loose cut 40).
\end{center}
\begin{physpicture}
The measured non-monotonic dispersion curve is therefore \textbf{windowed
physics}, not an intrinsic wavelength selection. This is the single most
counter-intuitive statement of the whole theory.
\end{physpicture}
\end{frame}

\begin{frame}{The peak: resonance surface meets the window}
\footnotesize
Resonance surface $\xi_{\rm res}\simeq(\kz-\gamma/\Vo)/(\alpha\Vo)+\ln2$ moves
outward \emph{linearly} with $\kz$. Setting $\xi_{\rm res}(\kz)=\xi_w$,
$\gamma\approx\Ceil$ at the crossover:
\[
\boxed{\;\kz^{\rm peak}\simeq\alpha\Vo(\xi_w-\ln2)+c\,(\alpha/\Vo)^{1/3},\qquad c=1.0\pm0.1\;}
\]
\textbf{Both terms $\eps$-independent}: coupling and containment radius select
the wavelength, together — not the shear width.
\centering\includegraphics[width=0.48\textwidth]{fig_peak.pdf}
\end{frame}

\begin{frame}{Roll-off, and why the data used to say ``$\kz^{\rm peak}\approx2\alpha$''}
Beyond the peak (window-truncated well, high-$\kz$ limit of Model A):
\[
\boxed{\;\gamma\simeq\sqrt{2\alpha\Vo/\kz}\;}\qquad(\gamma\propto\kz^{-1/2}\text{ decline, matches data})
\]
\begin{physpicture}
Over the historical survey grid, window radii shrank as $\alpha$ grew
($\alpha\Vo\xi_w\approx1$) — the sub-linear resonance term plus a shrinking window
term \emph{mimics} a linear trend across a factor-5 range of $\alpha$.
``$\kz^{\rm peak}\approx2\alpha$'' is a serviceable \emph{survey-box mnemonic}; the
boxed formula above is the actual law.
\end{physpicture}
\end{frame}

\section{The quantization budget: how $\eps$ actually enters}

\begin{frame}{Nondimensionalizing the action}
\footnotesize
Neither the ceiling nor the peak formula contains $\eps$ — where does it enter?
Changing variables $x=\eps\xi$, $w=\kz-u$, scaling out the drive:
\[
\boxed{\;S=\frac{\eps}{\Vo\,(\alpha\Vo^2)^{1/3}}\;\hat I(\hat k,\hat\gamma),
\qquad \hat I=\int_{\hat w\text{-well}}\sqrt{\hat Q}\,d\hat w\;}
\]
$\hat I$ is \emph{universal} — $\eps$-free. The shear width enters the eigenvalue
problem \textbf{only} as a multiplicative budget on the action.
\centering\includegraphics[width=0.32\textwidth]{fig_budget.pdf}
\end{frame}

\begin{frame}{The $P$-group: $\eps$ buys action}
Larger $\eps$ (fixed window, physics) enlarges $S$ at given $\gamma$; since $S$
decreases in $\gamma$, quantization pushes $\gamma$ \emph{up} toward the ceiling —
the sign of the measured 10--39\% rise of $\gamma$ from $\eps=0.10\to0.45$.
\[
P\equiv\frac{(\alpha\Vo^2)^{1/3}}{\eps}\qquad\text{(the ``quantization budget'')}
\]
\begin{physpicture}
$\eps$ buys action, not a wavelength shift. Stretching the layer stretches the
well in physical $x$ while its depth/shape in $\hat w$ is unchanged — so at fixed
$\gamma$ the mode accumulates more phase, and the connection condition lets
$\gamma$ float up. This is why $\eps$'s effect is a fractional \emph{rise} of
$\gamma$ everywhere, not a shift of $\Ceil,\Kwn$ — and it's the theory this
talk's GPU section will confront directly (T1.1).
\end{physpicture}
\end{frame}

\begin{frame}{Cold two-stream rates (fixes the filter design)}

\;\Longrightarrow\;
\boxed{k\Vo<\sqrt2}\ \text{(unstable band)}
\]
\[

\]

\begin{frame}{Nonlinear theory I: the density cavitation threshold}
Why $\Vo\gtrsim0.08$ hits a hard wall: a growing compressible mode carries
$\delta n\propto p/\gamma$; cavitation ($n\to$ code floor) occurs at
\[
\boxed{\;p_{\rm cav}\sim\gamma/\max(\kz,k_x^{\rm eff})\;}
\]
\begin{physpicture}
An \emph{amplitude} cliff, not a time cliff. A growing wave in a compressible
fluid carries a density modulation $\propto$ momentum/$\gamma$; the quadratic
(ponderomotive) beat then pushes plasma out of high-field regions at $2\gamma$.
This explains why the failure zone maps to $\Vo$ (drive strength) and cares
little about window design or mechanism — and why the only safe practice at
high $\Vo$ is to stop once the linear plateau is banked.
\end{physpicture}
\end{frame}

\begin{frame}{Nonlinear theory II: the back-reaction on $\Az$}

\]

Sign \emph{deepens} the background — no depletion. Waves weigh on their
background like radiation pressure; the frozen-$\Az$ approximation is
vindicated, not just assumed — the last escape route for that caveat closes.
\end{physmeaning}
\end{frame}

\begin{frame}{Warm closure and eigenmode seeding}
